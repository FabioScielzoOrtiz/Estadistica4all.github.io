Mathematically groupby() operation could be formalized as follows: \(\\[0.7cm]\)

\[df.groupby( \hspace{0.1cm} name(X_3) \hspace{0.1cm} ).f() \hspace{0.1cm}=\hspace{0.1cm} \begin{pmatrix} f\hspace{0.1cm} ( \hspace{0.1cm} x_{i1} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=0 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & f \hspace{0.1cm} ( \hspace{0.1cm} x_{i2} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=0 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & ... & f \hspace{0.1cm} ( \hspace{0.1cm} x_{ip} \hspace{0.1cm}:\hspace{0.1cm} x_{i3}=0 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) \\ f \hspace{0.1cm} ( \hspace{0.1cm} x_{i1} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=1 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & f \hspace{0.1cm}( \hspace{0.1cm} x_{i2} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=1 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & ... & f \hspace{0.1cm} ( \hspace{0.1cm} x_{ip} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=1 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) \\ f \hspace{0.1cm} ( \hspace{0.1cm} x_{i1} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=2 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm}) & f \hspace{0.1cm} ( \hspace{0.1cm} x_{i2} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=2 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & ... & f \hspace{0.1cm} ( \hspace{0.1cm} x_{ip} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=2 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) \\ f \hspace{0.1cm} ( \hspace{0.1cm} x_{i1} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=3 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & f \hspace{0.1cm} ( \hspace{0.1cm} x_{i2} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=3 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) & ... & f \hspace{0.1cm} (\hspace{0.1cm} x_{ip} \hspace{0.1cm} : \hspace{0.1cm} x_{i3}=3 \hspace{0.1cm} , \hspace{0.1cm} i=1,...,n \hspace{0.1cm} ) \end{pmatrix} \\[2cm]\]
